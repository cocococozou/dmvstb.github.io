time_df <-df_combine %>% 
  na.omit(`_COL19`) %>% 
  rename(tbtime = `_COL19`) %>% 
  select(tbtime, tb, gender, dmage, birthyear) %>% 
  mutate(gender = as.numeric(gender),
         birthyear = as.numeric(birthyear), 
         dmage = as.numeric(dmage),
         tb = as.numeric(tb),
         tbtime = as.numeric(tbtime))
